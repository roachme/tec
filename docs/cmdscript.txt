# Initialize tec
tec init

# We create a testing environment, name it 'test' and add some tasks
tec env add test
tec add test1
tec add test2
tec add test3

# Let's see those tasks
tec ls

# Switch to user home directory
cd

# Switch back to current task directory
tec cd

# Switch to previous task
tec cd -

# Delete current task
tec rm -f

# User current directory is automatically switch to previous
pwd

# Current task directory is removed
tec ls
